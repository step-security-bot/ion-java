13  /**
14   * Monitors an InputStream over binary Ion data to ensure enough data is available to be navigated successfully by a
15   * non-incremental IonReader.
16   * <p>
17   * Error reporting: this wrapper reads the least amount of Ion data possible in order to determine whether a value
18   * is complete. As such, it will not raise any errors if invalid data exists anywhere outside the header of a
19   * top-level value. Any such invalid data will be detected as normal by the IonReader. In the few cases where this
20   * wrapper does detect an error (e.g. upon finding the illegal type 0xF), it will raise {@link IonException}.
21   */
22  public final class IonReaderLookaheadBuffer extends ReaderLookaheadBufferBase {
23  
24      private static final int LOWER_SEVEN_BITS_BITMASK = 0x7F;
25      private static final int HIGHEST_BIT_BITMASK = 0x80;
26      private static final int VALUE_BITS_PER_VARUINT_BYTE = 7;
27      // Note: because long is a signed type, Long.MAX_VALUE is represented in Long.SIZE - 1 bits.
28      private static final int MAXIMUM_SUPPORTED_VAR_UINT_BYTES = (Long.SIZE - 1) / VALUE_BITS_PER_VARUINT_BYTE;
29      private static final int IVM_START_BYTE = 0xE0;
30      private static final int IVM_REMAINING_LENGTH = 3; // Length of the IVM after the first byte.
31      private static final int ION_SYMBOL_TABLE_SID = 3;
32  
33      /**
34       * Represents a VarUInt that may be read in multiple steps.
35       */
36      private static final class VarUInt {
37  
38          /**
39           * The location of the VarUInt in the value header.
40           */
41          private enum Location {
42              /**
43               * The length field that is included when the low nibble of a value's type ID is VARIABLE_LENGTH_NIBBLE.
44               */
45              VALUE_LENGTH,
46              /**
47               * The length field that is included when the low nibble of an annotation wrapper's type ID is
48               * VARIABLE_LENGTH_NIBBLE.
49               */

56              /**
57               * A symbol ID. An annotation wrapper may contain more than one.
58               */
59              ANNOTATION_WRAPPER_SID

79           * True when the VarUInt is complete; otherwise, false.
80           */
81          private boolean isComplete;

85           */

90          /**
91           * Resets the value to zero.
92           * @param nextLocation the location of the next VarUInt to read.
93           */
94          private void reset(final Location nextLocation) {
95              location = nextLocation;
96              value = 0;
97              numberOfBytesRead = 0;
98              isComplete = false;
99          }

107          * Positioned before the type ID of a top-level value.

299     /**
300      * Constructs a wrapper with the given configuration.
301      * @param configuration the configuration for the new instance.
302      * @param inputStream an InputStream over binary Ion data.
303      */
304     public IonReaderLookaheadBuffer(final IonBufferConfiguration configuration, final InputStream inputStream) {
305         super(configuration, inputStream);
306         pipe.registerNotificationConsumer(
307             new ResizingPipedInputStream.NotificationConsumer() {
308                 @Override
309                 public void bytesConsolidatedToStartOfBuffer(int leftShiftAmount) {
310                     // The existing data in the buffer has been shifted to the start. Adjust the saved indexes
311                     // accordingly. -1 indicates that all indices starting at 0 will be shifted.
312                     shiftIndicesLeft(-1, leftShiftAmount);
313                 }
314             }
315         );
316         pageSize = configuration.getInitialBufferSize();
317         oversizedSymbolTableHandler = configuration.getOversizedSymbolTableHandler();
318         inProgressVarUInt = new VarUInt();
319         reset();
320     }

331     /**
332      * Reads one byte, if possible.
333      * @return the byte, or -1 if none was available.
334      * @throws Exception if thrown by a handler method or if an IOException is thrown by the underlying InputStream.
335      */
336     private int readByte() throws Exception {
337         if (pipe.availableBeyondBoundary() == 0 && fillPage(1) < 1) {
338             return -1;
339         }
340         int b;
341         if (isSkippingCurrentValue()) {
342             // If the value is being skipped, the byte will not have been buffered.
343             b = getInput().read();
344         } else {
345             b = pipe.peek(peekIndex);
346             pipe.extendBoundary(1);
347             peekIndex++;
348         }
349         return b;
350     }

352     /**
353      * Reads a VarUInt. NOTE: the VarUInt must fit in a `long`. This is not a true limitation, as IonJava requires
354      * VarUInts to fit in an `int`.
355      * @throws Exception if thrown by a handler method or if an IOException is thrown by the underlying InputStream.
356      */
357     private void readVarUInt() throws Exception {
358         int currentByte;
359         while (inProgressVarUInt.numberOfBytesRead < MAXIMUM_SUPPORTED_VAR_UINT_BYTES) {
360             currentByte = readByte();
361             if (currentByte < 0) {
362                 return;
363             }
364             inProgressVarUInt.numberOfBytesRead++;
365             inProgressVarUInt.value =
366                     (inProgressVarUInt.value << VALUE_BITS_PER_VARUINT_BYTE) | (currentByte & LOWER_SEVEN_BITS_BITMASK);
367             if ((currentByte & HIGHEST_BIT_BITMASK) != 0) {
368                 inProgressVarUInt.isComplete = true;
369                 dataHandler.onData(inProgressVarUInt.numberOfBytesRead);
370                 return;
371             }
372         }
373         throw new IonException("Found a VarUInt that was too large to fit in a `long`");
374     }

396         STRUCT,

399          */
400         NOT_STRUCT,

403          * again when more data is available.
404          */
405         NO_DATA

408     /**
409      * Reads the type ID byte.
410      * @param isUnannotated true if this type ID is not on a value within an annotation wrapper; false if it is.
411      * @return the result as a {@link ReadTypeIdResult}.
412      * @throws Exception if thrown by a handler method or if an IOException is thrown by the underlying InputStream.
413      */
414     private ReadTypeIdResult readTypeID(final boolean isUnannotated) throws Exception {
415         int header = readByte();
416         if (header < 0) {
417             return ReadTypeIdResult.NO_DATA;
418         }
419         valueTid = IonTypeID.TYPE_IDS[header];
420         dataHandler.onData(1);
421         if (header == IVM_START_BYTE) {
422             if (!isUnannotated) {
423                 throw new IonException("Invalid annotation header.");
424             }
425             additionalBytesNeeded = IVM_REMAINING_LENGTH;
426             isSystemValue = true;
427             // Encountering an IVM resets the symbol table context; no need to parse any previous symbol tables.
428             resetSymbolTableMarkers();
429             ivmSecondByteIndex = peekIndex;
430             state = State.SKIPPING_VALUE;
431         } else if (!valueTid.isValid) {
432             throw new IonException("Invalid type ID.");
433         } else if (valueTid.type == IonType.BOOL) {
434             // bool values are always a single byte.
435             state = State.BEFORE_TYPE_ID;
436         } else if (valueTid.type == IonTypeID.ION_TYPE_ANNOTATION_WRAPPER) {
437             // Annotation.
438             if (valueTid.variableLength) {
439                 initializeVarUInt(VarUInt.Location.ANNOTATION_WRAPPER_LENGTH);
440             } else {
441                 setAdditionalBytesNeeded(valueTid.length, isUnannotated);
442                 initializeVarUInt(VarUInt.Location.ANNOTATION_WRAPPER_SIDS_LENGTH);
443             }
444         } else {
445             if (valueTid.isNull) {
446                 // null values are always a single byte.
447                 state = State.BEFORE_TYPE_ID;
448             } else {
449                 // Not null
450                 if (valueTid.variableLength) {
451                     initializeVarUInt(VarUInt.Location.VALUE_LENGTH);
452                 } else {
453                     setAdditionalBytesNeeded(valueTid.length, isUnannotated);
454                     state = State.SKIPPING_VALUE;
455                 }
456             }
457         }
458         if (valueTid.type == IonType.STRUCT) {
459             return ReadTypeIdResult.STRUCT;
460         }
461         return ReadTypeIdResult.NOT_STRUCT;
462     }

464     /**
465      * Reads the bytes of the value header that occur after the type ID byte and before the first value byte.
466      * @throws Exception if thrown by a handler method or if an IOException is thrown by the underlying InputStream.
467      */
468     private void readHeader() throws Exception {
469         if (inProgressVarUInt.location == VarUInt.Location.VALUE_LENGTH) {
470             readVarUInt();
471             if (inProgressVarUInt.isComplete) {
472                 additionalBytesNeeded = inProgressVarUInt.value;
473                 state = State.SKIPPING_VALUE;
474             }
475             return;
476         }
477         if (inProgressVarUInt.location == VarUInt.Location.ANNOTATION_WRAPPER_LENGTH) {
478             readVarUInt();
479             if (!inProgressVarUInt.isComplete) {
480                 return;
481             }
482             additionalBytesNeeded = inProgressVarUInt.value;
483             initializeVarUInt(VarUInt.Location.ANNOTATION_WRAPPER_SIDS_LENGTH);
484         }
485         if (inProgressVarUInt.location == VarUInt.Location.ANNOTATION_WRAPPER_SIDS_LENGTH) {
486             readVarUInt();
487             if (!inProgressVarUInt.isComplete) {
488                 return;
489             }
490             additionalBytesNeeded -= inProgressVarUInt.numberOfBytesRead;
491             numberOfAnnotationSidBytesRemaining = inProgressVarUInt.value;
492             initializeVarUInt(VarUInt.Location.ANNOTATION_WRAPPER_SID);
493         }
494         if (inProgressVarUInt.location == VarUInt.Location.ANNOTATION_WRAPPER_SID) {
495             while (true) {
496                 readVarUInt();
497                 if (inProgressVarUInt.isComplete) {
498                     currentNumberOfAnnotations++;
499                     if (currentNumberOfAnnotations == 1 && inProgressVarUInt.value == ION_SYMBOL_TABLE_SID) {
500                         isSymbolTableAnnotationFirst = true;
501                     }
502                     annotationSids.add((int) inProgressVarUInt.value);
503                     numberOfAnnotationSidBytesRemaining -= inProgressVarUInt.numberOfBytesRead;
504                     additionalBytesNeeded -= inProgressVarUInt.numberOfBytesRead;
505                     if (numberOfAnnotationSidBytesRemaining <= 0) {
506                         state = State.SKIPPING_VALUE;
507                     } else {
508                         initializeVarUInt(VarUInt.Location.ANNOTATION_WRAPPER_SID);
509                         continue;
510                     }
511                     if (isSymbolTableAnnotationFirst) {
512                         state = State.READING_VALUE_WITH_SYMBOL_TABLE_ANNOTATION;
513                     }
514                 }
515                 break;
516             }
517         }
518     }

520     /**
521      * Shift all indices after 'afterIndex' left by the given amount. This is used when data is moved in the underlying
522      * buffer either due to buffer growth or NOP padding being reclaimed to make room for a value that would otherwise
523      * exceed the buffer's maximum size.
524      * @param afterIndex all indices after this index will be shifted (-1 indicates that all indices should be shifted).
525      * @param shiftAmount the amount to shift left.
526      */
527     private void shiftIndicesLeft(int afterIndex, int shiftAmount) {
528         peekIndex = Math.max(peekIndex - shiftAmount, 0);
529         valuePreHeaderIndex -= shiftAmount;
530         valuePostHeaderIndex -= shiftAmount;
531         for (SymbolTableMarker symbolTableMarker : symbolTableMarkers) {
532             if (symbolTableMarker.startIndex > afterIndex) {
533                 symbolTableMarker.startIndex -= shiftAmount;
534                 symbolTableMarker.endIndex -= shiftAmount;
535             }
536         }
537         if (ivmSecondByteIndex > afterIndex) {
538             ivmSecondByteIndex -= shiftAmount;
539         }
540     }

542     /**
543      * Reclaim the NOP padding that occurred before the current value, making space for the value in the buffer.
544      */
545     private void reclaimNopPadding() {
546         pipe.consolidate(valuePreHeaderIndex, nopPadStartIndex);
547         shiftIndicesLeft(nopPadStartIndex, valuePreHeaderIndex - nopPadStartIndex);
548         resetNopPadIndex();
549     }

551     /**
552      * Skip bytes from the underlying InputStream without ever buffering them.
553      * @param numberOfBytesToSkip the number of bytes to attempt to skip.
554      * @return the number of bytes actually skipped.
555      * @throws IOException if thrown by the underlying InputStream.
556      */
557     private int skipBytesFromInput(int numberOfBytesToSkip) throws IOException {
558         try {
559             return (int) getInput().skip(numberOfBytesToSkip);
560         } catch (EOFException e) {
561             // Certain InputStream implementations (e.g. GZIPInputStream) throw EOFException if more bytes are requested
562             // to skip than are currently available (e.g. if a header or trailer is incomplete).
563             return 0;
564         }
565     }

567     /**
568      * Retrieve and buffer up to {@link #pageSize} bytes from the input.
569      * @param numberOfBytesRequested the minimum amount of space that must be available before the buffer reaches
570      *                               its configured maximum size.
571      * @return the number of bytes buffered by this operation.
572      * @throws Exception if thrown by the underlying InputStream.
573      */
574     private int fillPage(int numberOfBytesRequested) throws Exception {
575         int amountToFill = pipe.capacity() - pipe.size();
576         if (amountToFill <= 0) {
577             // Try to fill the remainder of the existing buffer to avoid growing unnecessarily. If there is no
578             // space, that indicates that a single value exceeds the size of a page. In that case, fill another page,
579             // growing the buffer only up to the configured maximum size.
580             int spaceAvailable = getMaximumBufferSize() - pipe.capacity();
581             if (numberOfBytesRequested > spaceAvailable) {
582                 if (nopPadStartIndex > -1 && valuePreHeaderIndex - nopPadStartIndex >= numberOfBytesRequested) {
583                     // Reclaim the NOP pad space if doing so would allow the value to fit.
584                     reclaimNopPadding();
585                 } else {
586                     startSkippingValue();
587                 }
588                 amountToFill = numberOfBytesRequested;
589             } else {
590                 amountToFill = Math.min(pageSize, spaceAvailable);
591             }
592         }
593         int received;
594         if (isSkippingCurrentValue()) {
595             if (state == State.SKIPPING_VALUE) {
596                 // This is a seek operation, meaning that the bytes don't need to be interpreted.
597                 received = skipBytesFromInput(amountToFill);
598             } else {
599                 // The bytes need to be interpreted, so they cannot be skipped. The caller must retrieve them from
600                 // the input.
601                 received = amountToFill;
602             }
603         } else {
604             received = pipe.receive(getInput(), amountToFill);
605         }
606         return received;
607     }

609     /**
610      * Notifies the event handler that the current value is oversized, if the handler has not already been notified.
611      * @throws Exception if thrown by the handler.
612      */
613     private void notifyHandlerOfOversizedValue() throws Exception {
614         if (handlerNeedsToBeNotifiedOfOversizedValue) {
615             if (isSystemValue) {
616                 // Processing cannot continue after system values (symbol tables) are truncated because subsequent
617                 // values may be unreadable. Notify the user.
618                 oversizedSymbolTableHandler.onOversizedSymbolTable();
619             } else {
620                 // An oversized user value has been encountered. Notify the user so they can decide whether to continue
621                 // or abort.
622                 oversizedValueHandler.onOversizedValue();
623             }
624         }
625         handlerNeedsToBeNotifiedOfOversizedValue = false;
626     }

628     /**
629      * Attempt to retrieve at least `additionalBytesNeeded` bytes from the input and either buffer them (if the value
630      * is being consumed) or skip them (if the value is being skipped due to being oversize).
631      * @return the number of bytes filled or skipped.
632      * @throws Exception if thrown by the event handler.
633      */
634     private int fillOrSkip() throws Exception {
635         // Clamping at the number of buffered bytes available guarantees that the buffer
636         // will never grow beyond its initial size.
637         int bytesRequested = (int) additionalBytesNeeded - pipe.availableBeyondBoundary();
638         int bytesFilled;
639         if (isSkippingCurrentValue()) {
640             bytesFilled = skipBytesFromInput(bytesRequested);
641         } else {
642             bytesFilled = fillPage(bytesRequested);
643         }
644         if (bytesFilled < 1) {
645             return 0;
646         }
647         if (isSkippingCurrentValue()) {
648             // The user cannot be notified of a size violation until it has been determined whether
649             // the value is a symbol table or user value, which is only true in the SKIPPING_VALUE
650             // state.
651             notifyHandlerOfOversizedValue();
652             // Skip all of the bytes skipped from the InputStream as well as all bytes previously
653             // buffered.
654             bytesFilled = bytesFilled + ((int) additionalBytesNeeded - bytesRequested);
655         } else {
656             bytesFilled = (int) Math.min(additionalBytesNeeded, bytesFilled);
657             pipe.extendBoundary(bytesFilled);
658             peekIndex += bytesFilled;
659         }
660         return bytesFilled;
661     }

663     /*
664      * The state transitions of the fillInput() method are summarized by the following diagram.
665      *
666      *                                 fillInput()
667      *                                       |
668      *                                       |   +----------------------------------------------------+
669      *                                       |   |                                                    |
670      *       Read first byte of IVM +--------v---v-+                                                  |
671      *   +--------------------------+BEFORE_TYPE_ID<--+                                               |
672      *   |   Or length was inferred |              +--+1-byte value(null or bool) read                |
673      *   |    from the type ID    +>+------+-------+                                                  |
674      *   |                        |        |                                                          |
675      *   |                        |        |No bytes available                                        |
676      *   |           1-byte value |        v                                                          |
677      *   |                        | +----------------+                                                |
678      *   |                        +-+READING_TYPE_ID <-+                                              |
679      *   |                          |                +-+No bytes available                            |
680      *   |                          +------+---------+                                                |
681      *   |                                 |Read type-id of multi-byte value                          |
682      *   |                                 v                                                          |
683      *   |      Finished header     +----------------+                                                |
684      *   | +------------------------+READING_HEADER  <--+                                             |
685      *   | |                        |                +--+Not enough bytes to complete header          |
686      *   | |                        +------+---------+                                                |
687      *   | |                               |Read annotation wrapper with symbol table annotation      |
688      *   | |                               v                                                          |
689      *   | |             +---------------------------------------------+                              |
690      *   | |             |READING_VALUE_WITH_SYMBOL_TABLE_ANNOTATION   <--+                           |
691      *   | |             |                                             +--+No bytes available         |
692      *   | |             +--+--------------+---------------------------+                              |
693      *   | |                |              |The wrapped value is a struct                             |
694      *   | | The wrapped    |              v                                                          |
695      *   | |  value is not  |  +-----------------------------+                                        |
696      *   | |   a struct     |  | READING_SYMBOL_TABLE_LENGTH <--+                                     |
697      *   | |                |  |                             +--+Not enough bytes to complete length  |
698      *   | |                |  +-----------+-----------------+                                        |
699      *   | |                |              |Read length                                               |
700      *   | |                |              v                                                          |
701      *   | |                |      +-----------------+                                                |
702      *   | +--------------->+----->| SKIPPING_VALUE  <--+                                             |
703      *   |                         |                 +--+More bytes needed to complete skip           |
704      *   +------------------------>+-------+---------+                                                |
705      *                                     |                                                          |
706      *                                     |All bytes skipped                                         |
707      *                                     |                                                          |
708      *                                     +----------------------------------------------------------+
709      */
710     @Override
711     protected void fillInputHelper() throws Exception {
712         while (true) {
713             if (state == State.BEFORE_TYPE_ID || state == State.READING_TYPE_ID) {
714                 reset();
715                 state = State.READING_TYPE_ID;
716                 if (readTypeID(true) != ReadTypeIdResult.NO_DATA) {
717                     // The previous line transfers at most one byte, so the pre-header index is the write index minus
718                     // one.
719                     valuePostHeaderIndex = peekIndex;
720                     valuePreHeaderIndex = valuePostHeaderIndex - 1;
721                     valueStartWriteIndex = valuePreHeaderIndex;
722                 }
723             }
724             if (state == State.READING_HEADER) {
725                 readHeader();
726                 if (!inProgressVarUInt.isComplete) {
727                     return;
728                 }
729                 valuePostHeaderIndex = peekIndex;
730             }
731             if (state == State.READING_VALUE_WITH_SYMBOL_TABLE_ANNOTATION) {
732                 ReadTypeIdResult result = readTypeID(false);
733                 if (result == ReadTypeIdResult.NO_DATA) {
734                     return;
735                 }
736                 // When successful, readTypeID reads exactly one byte.
737                 additionalBytesNeeded--;
738                 if (result == ReadTypeIdResult.STRUCT) {
739                     state = State.READING_SYMBOL_TABLE_LENGTH;
740                 } else {
741                     state = State.SKIPPING_VALUE;
742                 }
743             }
744             if (state == State.READING_SYMBOL_TABLE_LENGTH) {
745                 isSystemValue = true;
746                 if (inProgressVarUInt.location == VarUInt.Location.VALUE_LENGTH) {
747                     readVarUInt();
748                     if (!inProgressVarUInt.isComplete) {
749                         return;
750                     }
751                     additionalBytesNeeded = inProgressVarUInt.value;
752                 }
753                 symbolTableMarkers.add(new SymbolTableMarker(peekIndex, (int) additionalBytesNeeded));
754                 state = State.SKIPPING_VALUE;
755             }
756             if (state == State.SKIPPING_VALUE) {
757                 if (valueTid.isNopPad) {
758                     if (pipe.availableBeyondBoundary() <= additionalBytesNeeded) {
759                         // There cannot be any meaningful data beyond the NOP pad, so the buffer can be truncated
760                         // immediately and the rest of the NOP pad skipped.
761                         additionalBytesNeeded -= pipe.availableBeyondBoundary();
762                         startSkippingValue();
763                         // NOP padding will not be buffered, so it is never considered oversized.
764                         handlerNeedsToBeNotifiedOfOversizedValue = false;
765                     }
766                     // Else, the rest of the NOP pad is already buffered, and there is a value at least partially
767                     // buffered beyond it. The NOP pad will only be deleted from the buffer if the next value is
768                     // large enough that it doesn't fit within the buffer's configured maximum size.
769                 }
770                 while (additionalBytesNeeded > 0) {
771                     int numberOfBytesToRead;
772                     if (pipe.availableBeyondBoundary() >= additionalBytesNeeded) {
773                         numberOfBytesToRead = (int) additionalBytesNeeded;
774                         pipe.extendBoundary(numberOfBytesToRead);
775                         peekIndex += numberOfBytesToRead;
776                     } else {
777                         numberOfBytesToRead = fillOrSkip();
778                         if (numberOfBytesToRead < 1) {
779                             return;
780                         }
781                     }
782                     dataHandler.onData(numberOfBytesToRead);
783                     additionalBytesNeeded -= numberOfBytesToRead;
784                 }
785                 state = State.BEFORE_TYPE_ID;
786             }
787             if (state == State.BEFORE_TYPE_ID) {
788                 valueEndIndex = peekIndex;
789                 if (isSystemValue || isSkippingCurrentValue() || valueTid.isNopPad) {
790                     if (valueTid.isNopPad && nopPadStartIndex < 0) {
791                         // This is the first NOP before the next value. Mark the start index in case the space needs to
792                         // be reclaimed later.
793                         nopPadStartIndex = valuePreHeaderIndex;
794                     }
795                     if (isSystemValue && isSkippingCurrentValue()) {
796                         // The symbol table(s) currently buffered exceed the maximum buffer size. This is not
797                         // recoverable; future invocations of fillInput() will do nothing.
798                         reset();
799                         state = State.DONE;
800                     } else {
801                         if (isSystemValue && nopPadStartIndex > -1) {
802                             // Reclaim any NOP pad space that precedes system values. This will usually not be strictly
803                             // necessary, but it simplifies the implementation and will be rare in practice. Without
804                             // this simplification, we would need to keep track of a list of NOP pad start/end indexes
805                             // as we do with the symbol table markers. This way, we know that there can only be one
806                             // uninterrupted run of NOP pad bytes immediately preceding any user value, making it easy
807                             // to reclaim this space if necessary.
808                             reclaimNopPadding();
809                         }
810                         // Just skipped over system value or an oversized value. Consume the next value too so that a
811                         // call to reader.next() won't return null.
812                         continue;
813                     }
814                 }
815             }
816             break;
817         }
818     }

821     void truncateToEndOfPreviousValue() {
822         peekIndex = valueStartWriteIndex;
823         pipe.truncate(valueStartWriteIndex, valueStartAvailable);
824         handlerNeedsToBeNotifiedOfOversizedValue = true;
825     }

832     /**
833      * Rewinds to the start of the value currently buffered. Does not include any system values that may precede
834      * the value. This method is not called in conjunction with {@link #mark()} / {@link #rewind()}, which may be
835      * used if the caller wants to rewind to the start of any system values that precede the current value. This
836      * method may be used to re-read the current value and may only be called after {@code IonReader.next()}
837      * has been called on the current value; otherwise, the data representing any system values that precede the
838      * current value would be lost.
839      *
840      * @throws IllegalStateException if there is no value currently buffered or if system value data would be lost
841      *   as a result of calling this method before {@code IonReader.next()} was called.
842      */
843     public void rewindToValueStart() {
844         if (valuePreHeaderIndex < 0) {
845             throw new IllegalStateException("A value must be buffered before calling rewindToValueStart().");
846         }
847         int availableAtValueStart = pipe.getBoundary() - valuePreHeaderIndex;
848         // If rewinding would reduce the amount of data available, that indicates that system value data would be lost.
849         if (availableAtValueStart < available()) {
850             throw new IllegalStateException(
851                 "IonReader.next() must be called on the current value before calling rewindToValueStart()."
852             );
853         }
854         pipe.rewind(valuePreHeaderIndex, availableAtValueStart);
855         peekIndex = valuePreHeaderIndex;
856     }
