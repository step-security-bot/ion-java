1   package com.amazon.ion.impl;
2   
3   import com.amazon.ion.IonType;
4   
5   /**
6    * Holds pre-computed information about a binary Ion type ID byte.
7    */
8   final class IonTypeID {
9   
10      private static final int NUMBER_OF_BYTES = 0x100;
11      private static final int BITS_PER_NIBBLE = 4;
12      private static final int LOW_NIBBLE_BITMASK = 0x0F;
13      private static final int NULL_VALUE_NIBBLE = 0xF;
14      private static final int VARIABLE_LENGTH_NIBBLE = 0xE;
15      private static final int NEGATIVE_INT_TYPE_CODE = 0x3;
16      private static final int TYPE_CODE_INVALID = 0xF;
17      private static final int ANNOTATION_WRAPPER_MIN_LENGTH = 0x3;
18      private static final int ANNOTATION_WRAPPER_MAX_LENGTH = 0xE;
19      static final int ORDERED_STRUCT_NIBBLE = 0x1;
20  
21      // NOTE: 'annotation wrapper' is not an IonType, but it is simplest to treat it as one for the purposes of this
22      // implementation in order to have a direct mapping from binary type IDs to IonType enum values. IonType.DATAGRAM
23      // does not have a type ID, so we will use it to mean 'annotation wrapper' instead.
24      static final IonType ION_TYPE_ANNOTATION_WRAPPER = IonType.DATAGRAM;
25  
26      // Lookup table from type ID to IonType. See https://amzn.github.io/ion-docs/docs/binary.html#typed-value-formats
27      static final IonTypee">IonType[] ION_TYPES = new IonType[] {
28          IonType.NULL,
29          IonType.BOOL,
30          IonType.INT,
31          IonType.INT,
32          IonType.FLOAT,
33          IonType.DECIMAL,
34          IonType.TIMESTAMP,
35          IonType.SYMBOL,
36          IonType.STRING,
37          IonType.CLOB,
38          IonType.BLOB,
39          IonType.LIST,
40          IonType.SEXP,
41          IonType.STRUCT,
42          ION_TYPE_ANNOTATION_WRAPPER,
43          null // The 0xF type code is illegal in Ion 1.0.
44      };
45  
46      // Pre-compute all possible type ID bytes.
47      static final IonTypeID[] TYPE_IDS;
48      static {
49          TYPE_IDS = new IonTypeID[NUMBER_OF_BYTES];
50          for (int b = 0x00; b < NUMBER_OF_BYTES; b++) {
51              TYPE_IDS[b] = new IonTypeID((byte) b);
52          }
53      }
54  
55      final IonType type;
56      final byte length;
57      final boolean variableLength;
58      final boolean isNull;
59      final boolean isNopPad;
60      final byte lowerNibble;
61      final boolean isValid;
62      final boolean isNegativeInt;
63  
64      /**
65       * Determines whether the Ion spec allows this particular upperNibble/lowerNibble pair.
66       */
67      private static boolean isValid(byte upperNibble, byte lowerNibble, IonType type) {
68          if (upperNibble == TYPE_CODE_INVALID) {
69              // Type code F is unused in Ion 1.0.
70              return false;
71          }
72          if (type == IonType.BOOL) {
73              // Bool values can only be false (0), true (1), or null (F).
74              return lowerNibble <= 1 || lowerNibble == NULL_VALUE_NIBBLE;
75          }
76          if (type == IonType.INT && upperNibble == NEGATIVE_INT_TYPE_CODE) {
77              // There is no negative zero int.
78              return lowerNibble != 0;
79          }
80          if (type == IonType.FLOAT) {
81              // Floats are either 0e0 (0), 32-bit (4), 64-bit (8), or null (F).
82              return lowerNibble == 0 || lowerNibble == 4 || lowerNibble == 8 || lowerNibble == NULL_VALUE_NIBBLE;
83          }
84          if (type == IonType.TIMESTAMP) {
85              // There is no zero-length timestamp representation.
86              return lowerNibble > 1;
87          }
88          if (type == ION_TYPE_ANNOTATION_WRAPPER) {
89              return lowerNibble >= ANNOTATION_WRAPPER_MIN_LENGTH && lowerNibble <= ANNOTATION_WRAPPER_MAX_LENGTH;
90          }
91          return true;
92      }
93  
94      private IonTypeID(byte id) {
95          byte upperNibble = (byte) ((id >> BITS_PER_NIBBLE) & LOW_NIBBLE_BITMASK);
96          this.lowerNibble = (byte) (id & LOW_NIBBLE_BITMASK);
97          this.type = ION_TYPES[upperNibble];
98          this.isValid = isValid(upperNibble, lowerNibble, type);
99          this.isNull = lowerNibble == NULL_VALUE_NIBBLE;
100         this.isNopPad = type == IonType.NULL && !isNull;
101         byte length = lowerNibble;
102         if ((type == IonType.NULL && !isNopPad) || type == IonType.BOOL || !isValid) {
103             variableLength = false;
104             length = 0;
105         } else if (type == IonType.STRUCT && length == ORDERED_STRUCT_NIBBLE) {
106             variableLength = true;
107         } else {
108             variableLength = length == VARIABLE_LENGTH_NIBBLE;
109         }
110         if (isNull) {
111             length = 0;
112         }
113         this.isNegativeInt = type == IonType.INT && upperNibble == NEGATIVE_INT_TYPE_CODE;
114         this.length = length;
115     }
116 }
