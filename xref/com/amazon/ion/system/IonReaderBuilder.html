172     /**
173      * <p>
174      * Determines whether the IonReader will allow incremental reading of binary Ion data. When enabled, if
175      * {@link IonReader#next()} returns {@code null} at the top-level, it indicates that there is not enough data
176      * in the stream to complete a top-level value. The user may wait for more data to become available in the stream
177      * and call {@link IonReader#next()} again to continue reading. Unlike the non-incremental reader, the incremental
178      * reader will never throw an exception due to unexpected EOF during {@code next()}. If, however,
179      * {@link IonReader#close()} is called when an incomplete value is buffered, an {@link IonException} will be raised.
180      * </p>
181      * <p>
182      * There is currently no incremental text IonReader, so for text data a non-incremental IonReader will be
183      * returned regardless of the value of this option. If incremental text reading is supported in the future, it
184      * may be enabled via this option.
185      * </p>
186      * <p>
187      * When this option is enabled, auto-detection of GZIP data is not supported; the byte array or InputStream
188      * implementation provided to {@link #build(byte[])} or {@link #build(InputStream)} must return uncompressed bytes.
189      * This can be achieved by wrapping the data in a GZIPInputStream and passing it to {@link #build(InputStream)}.
190      * </p>
191      * <p>
192      * Additionally, when this option is enabled, annotation iterators are reused by default, improving performance.
193      * See {@link #withAnnotationIteratorReuseEnabled(boolean)} for more information and to disable that option.
194      * </p>
195      * <p>
196      * Although the incremental binary IonReader provides performance superior to the non-incremental binary IonReader
197      * for both incremental and non-incremental use cases, there is one caveat: the incremental implementation
198      * must be able to buffer an entire top-level value and any preceding system values (Ion version marker(s) and
199      * symbol table(s)) in memory. This will not be a problem for the vast majority of Ion streams, as it is rare for a
200      * single top-level value or symbol table to exceed a few megabytes in size. However, if the size of the stream's
201      * values risk exceeding the available memory, then this option must not be enabled.
202      * </p>
203      * @param isEnabled true if the option is enabled; otherwise, false.
204      *
205      * @return this builder instance, if mutable;
206      * otherwise a mutable copy of this builder.
207      *
208      * @see #setIncrementalReadingEnabled()
209      * @see #setIncrementalReadingDisabled()
210      */
211     public IonReaderBuilder withIncrementalReadingEnabled(boolean isEnabled) {
212         IonReaderBuilder b = mutable();
213         if (isEnabled) {
214             b.setIncrementalReadingEnabled();
215         } else {
216             b.setIncrementalReadingDisabled();
217         }
218         return b;
219     }

359     /**
360      * Based on the builder's configuration properties, creates a new IonReader
361      * instance over the given block of Ion data, detecting whether it's text or
362      * binary data.
363      * <p>
364      * This method will auto-detect and uncompress GZIPped Ion data.
365      *
366      * @param ionData the source of the Ion data, which is used only within the
367      * range of bytes starting at {@code offset} for {@code len} bytes.
368      * The data in that range may be either Ion binary data or UTF-8 Ion text.
369      * The reader retains a reference to the array, so its data must not be
370      * modified while the reader is active. Must not be null.
371      * @param offset must be non-negative and less than {@code ionData.length}.
372      * @param length must be non-negative and {@code offset+length} must not
373      * exceed {@code ionData.length}.
374      *
375      * @see IonSystem#newReader(byte[], int, int)
376      */
377     public IonReader build(byte[] ionData, int offset, int length)
378     {
379         if (isIncrementalReadingEnabled) {
380             if (IonStreamUtils.isGzip(ionData, offset, length)) {
381                 throw new IllegalArgumentException("Automatic GZIP detection is not supported with incremental" +
382                         "support enabled. Wrap the bytes with a GZIPInputStream and call build(InputStream).");
383             }
384             if (IonStreamUtils.isIonBinary(ionData, offset, length)) {
385                 return makeIncrementalReader(this, new ByteArrayInputStream(ionData, offset, length));
386             }
387         }
388         return makeReader(validateCatalog(), ionData, offset, length);
389     }

391     /**
392      * Based on the builder's configuration properties, creates a new IonReader
393      * instance over the given stream of Ion data, detecting whether it's text or
394      * binary data.
395      * <p>
396      * This method will auto-detect and uncompress GZIPped Ion data.
397      * <p>
398      * Because this library performs its own buffering, it's recommended that
399      * users avoid adding additional buffering to the given stream.
400      *
401      * @param ionData the source of the Ion data, which may be either Ion binary
402      * data or UTF-8 Ion text. Must not be null.
403      *
404      * @return a new reader instance.
405      * Callers must call {@link IonReader#close()} when finished with it.
406      *
407      * @throws IonException if the source throws {@link IOException}.
408      *
409      * @see IonSystem#newReader(InputStream)
410      */
411     public IonReader build(InputStream ionData)
412     {
413         InputStream wrapper = ionData;
414         if (isIncrementalReadingEnabled) {
415             if (!ionData.markSupported()) {
416                 wrapper = new BufferedInputStream(ionData);
417             }
418             wrapper.mark(_Private_IonConstants.BINARY_VERSION_MARKER_SIZE);
419             byte[] possibleIVM = new byte[_Private_IonConstants.BINARY_VERSION_MARKER_SIZE];
420             int bytesRead;
421             try {
422                 bytesRead = wrapper.read(possibleIVM);
423                 wrapper.reset();
424             } catch (IOException e) {
425                 throw new IonException(e);
426             }
427             if (IonStreamUtils.isGzip(possibleIVM, 0, possibleIVM.length)) {
428                 throw new IllegalArgumentException("Automatic GZIP detection is not supported with incremental" +
429                         "support enabled. Wrap the bytes with a GZIPInputStream and call build(InputStream).");
430             }
431             if (possibleIVM.length == bytesRead && IonStreamUtils.isIonBinary(possibleIVM)) {
432                 return makeIncrementalReader(this, wrapper);
433             }
434         }
435         return makeReader(validateCatalog(), wrapper);
436     }
